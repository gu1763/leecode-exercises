//c
void moveZeroes(int* nums, int numsSize) 
{
    int fast=0,slow=0;
    for(fast=0;fast<numsSize;fast++)
    {
        if(nums[fast]!=0)
        {
            nums[slow]=nums[fast];
            slow+=1;
        }
    }
    for(;slow<numsSize;slow++) nums[slow]=0;
}

//python3
class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        s=0
        for f in nums:
            if(f!=0):
                nums[s]=f
                s=s+1
        l=len(nums)
        while(s<l):
            nums[s]=0
            s=s+1
